### IRE Debug ###

country_event = {
       id = 40112
       title = "Debug"
       desc  = "Reduction of starting prestige"
       fire_only_once = yes

       trigger = {
               tag = IRE
	       NOT = { has_country_flag = prestige_debug }
       }

       option = {
              name = "Debug"
	      set_country_flag = prestige_debug
	      prestige = -300
       }
}
